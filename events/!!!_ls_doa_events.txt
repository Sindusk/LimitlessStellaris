	id = ascension_event.11
	title = ascension_event.11.name
	desc = ascension_event.11.desc
	hide_window = yes
	
	trigger = {
	    is_ai = yes
		is_subject = no
		num_owned_planets > 10
		num_ascension_perks > 7

		AND = {
			has_technology = tech_mega_engineering
			has_technology = tech_dark_matter_power_core
			has_technology = tech_ascension_theory
		}
		NOT = { 
		has_country_flag = ascension_started
		}
	}
	
	mean_time_to_happen = {
		years = 20
		modifier = {
			factor = 2.0
			AND = {
			    any_country = {
				   has_ascension_perk = ap_doa_galactic_ascendancy # LS changed to DoA's split version of the ascendancy perk.
				}
			}
		}
	}
	immediate = {
	        country_event = { id = ascension_event.1 }
	}
