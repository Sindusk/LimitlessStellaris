	potential = {
		is_capital = no

		}
	}

		planet_district_planet_storage_build_speed_mult = 1

			}

			}
            owner = { is_regular_empire = yes }
        }
        modifier = {
			job_yanz_machine_storage_drone_add = 2
        }
    }
}
