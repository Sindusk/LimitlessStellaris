### Extra Ship Components NEXT
# Disable planetary decision to open the mod menu.
decision_esc_mod_menu = {
	owned_planets_only = yes

	resources = {
		category = decisions
		cost = { energy = 0 }
	}

	potential = { always = no } # Disable through potential

	effect = {
		owner = {
			country_event = { id = esc_options.1000 days = 0 }
		}
	}
}

# Disable migration decisions.
PF_dec_heup_mig = {
	sound = event_administrative_work
	enactment_time = 1
	
	potential = { always = no } # Disable through potential
	allow = {
		from = { is_ai = no }
		hidden_trigger = { PF_trgr_plnt_heup_mig = yes }
		custom_tooltip = {
			fail_text = z_pf_tips_heup_mig_ignore
			hidden_trigger = { NOT = { has_planet_flag = PF_flg_plnt_mig_ignore } }
		}
	}
	
	effect = {
		custom_tooltip = z_pf_tips_heup_mig
		hidden_effect = { PF_eft_plnt_heup_mig = yes }

	ai_weight = { weight = 0 }
