## Trial by Fire [Advanced Policies]
# Disabled. Too strong for being so cheap. Also not a toggle.
AP_trial_by_fire = {
	owned_planets_only = yes
	sound = event_celebration
	icon = decision_alien_propaganda
	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}
	potential = {
		# LS - Disable through potential
		always = no
		#owner = {
		#	is_militarist = yes
		#}
	}
	allow = {
		planet = {
			NOT = {
				has_building = building_colony_shelter
				has_building = building_hab_capital
			}
			NOT = {
				has_modifier = planet_trial_by_fire
			}
		}
	}
	effect = {
		add_modifier = {
			modifier = planet_trial_by_fire
			days = @mediumlength
		}
	}
	ai_weight = {
		weight = 0.5
	}
}

## Create Life [Expanded Megastructures and Technology]
# Disabled. Adds a new species and leads to species bloat. Also creates new pops, which is starting to get phased out.
decision_create_life = {
	owned_planets_only = yes
	icon = decision_cybernetics
	prerequisites = {
		"tech_new_life"		
	}

	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}
	
	potential = {
		# LS - Disable through potential
		always = no
		#exists = owner
		#owner = {
		#	NOT = { has_country_flag = new_species_project }
		#	is_homicidal = no
		#}
	}
	
	effect = {
		custom_tooltip = create_life_tooltip
		hidden_effect = {
			set_planet_flag = new_species_planet
			owner = {
				country_event = { id = megatech.20 }
				if = {
					limit = { is_ai = yes }
					set_timed_country_flag = { flag = decision_create_life_timer days = 7200 }
				}
			}
		}
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			owner = { NOT = { has_country_flag = decision_create_life_timer } }
			free_housing >= 10
			free_jobs >= 5
			free_amenities >= 10
		}
	}
}
