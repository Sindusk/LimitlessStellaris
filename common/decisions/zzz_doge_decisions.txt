# Mining
decision_simple_planet_mining = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = minerals_designation

	resources = {
		category = decisions
		cost = {
			unity = 250
		}
	}

	effect = {
		add_modifier = { modifier = "simple_planet_mining" }
	}

	potential = {
		owner = {
			has_tradition = tr_industry_finish
		}
		NOT = { has_modifier = simple_planet_mining }
		NOT = { has_modifier = simple_planet_energy }
		NOT = { has_modifier = simple_planet_farmer }
		NOT = { has_modifier = simple_planet_artisan }
		NOT = { has_modifier = simple_planet_metallurgist }
		NOT = { has_modifier = simple_planet_unity }
		NOT = { has_modifier = simple_planet_trade }
	}

	allow = {
		NOT = { has_modifier = simple_planet_mining }
		NOT = { has_modifier = simple_planet_energy }
		NOT = { has_modifier = simple_planet_farmer }
		NOT = { has_modifier = simple_planet_artisan }
		NOT = { has_modifier = simple_planet_metallurgist }
		NOT = { has_modifier = simple_planet_unity }
		NOT = { has_modifier = simple_planet_trade }
	}

	ai_weight = {
		weight = 5
		modifier = {
			factor = 2
			OR = {
				has_designation = col_mining
				has_designation = col_habitat_mining
			}
		}
		modifier = {
			factor = 2
			num_minerals > 20
		}
		modifier = {
			factor = 2
			num_minerals > 40
		}
		modifier = {
			factor = 2
			num_minerals > 80
		}
		modifier = {
			factor = 2
			num_minerals > 200
		}
		modifier = {
			factor = 0
			OR = {
				NOT = {
					OR = {
						has_district = district_mining
						has_district = district_mining_uncapped
						has_district = district_hab_mining
					}
				num_pops < 30
				}
			}
		}
	}
}
