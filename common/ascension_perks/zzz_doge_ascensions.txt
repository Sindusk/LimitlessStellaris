## Great Navigator (Advanced Ascension Rework) gave 500% sublight speed.
# Disabled.
AAR_AP_navigator = {
	potential = {
		always = no
	}
	possible = {
		always = no
	}

	on_enabled = {
		custom_tooltip = "AAR_tips_AP_navigator"
		hidden_effect = {
			add_modifier = { modifier = AAR_MOD_cntr_navigator days = -1 }
		}
	}

	modifier = {
		pc_habitat_habitability = 0.05	#иор╩╫вн╙0.2

		ship_windup_mult = -0.35
		ship_winddown_mult = -0.35
	#	ship_ftl_jumpdrive_range_mult = 0.15
		ship_jumpdrive_cooldown_mult = -0.05
		
		ship_disengage_chance_mult = 0.25
		ship_emergency_ftl_min_days_mult = -0.50
		ship_emergency_ftl_mult = -0.25
	}

	ai_weight = {
		factor = 35
		modifier = {
			factor = 1.2
			OR = {
				has_ethic = ethic_fanatic_pacifist
				has_origin = origin_void_dwellers
			}
		}
		modifier = {
			factor = 1.2
			count_owned_planet = {
				count > 2
				limit = {
					is_planet_class = pc_habitat
				}
			}
		}
	}
}

## Imperial Prerogative (Vanilla) was underwhelming due to many other mods setting planet multiplier lower.
# Reduce empire size from planets from -50% to -20%
# Add empire size from systems -15%
# Add empire size from districts -10%
# Add empire size from pops -10%
ap_imperial_prerogative = {
    modifier = {
        empire_size_colonies_mult = -0.20
		empire_size_systems_mult = -0.15
		empire_size_districts_mult = -0.10
		empire_size_pops_mult = -0.10
    }
    potential = {
        NOT = {
            has_ascension_perk = ap_imperial_prerogative
        }
        NOT = {
            has_authority = auth_corporate
        }
    }
    ai_weight = {
        factor = 5
    }
    on_enabled = {
        run_ai_strategic_data = yes
    }
}
