	}

## Sanctum of the Whisperers (Tier 1)
# Merge between Sort Building, Real Space: Planetary Stations
building_whisperers_sanctum = {
	base_buildtime = @db3_time
	position_priority = 1021000 # Sort Buildings

	category = government

    # Building Tooltip
    triggered_desc = {
        text = building_whisperers_sanctum_tt
    }

	empire_limit = {
		base = 1
	}

	potential = {
		host_has_dlc = "Utopia"
		owner = { has_country_flag = whisperers_covenant_rank_1 }
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_planet_flag = planetary_station # Real Space: Planetary Stations
		}
	}

	allow = {
		has_major_upgraded_capital = yes
	}

	destroy_trigger = {
		exists = owner
		NOT = {
			owner = { has_country_flag = whisperers_covenant_rank_1 }
		}
	}

	planet_modifier = {
		job_telepath_add = 3
	}

	country_modifier = {
		envoys_add = 1
		spy_network_daily_value_mult = 0.15
	}

	triggered_desc = {
		trigger = {
			has_first_contact_dlc = yes
		}
		text = covenant_detection_first_contact_effect
	}

	on_built = {
		if = {
			limit = {
				has_first_contact_dlc = yes
			}
			planet = {
				solar_system = {
					starbase = {
						fleet = {
							add_modifier = {
								modifier = covenant_detection_first_contact
								days = -1
							}
						}
					}
				}
			}
		}
	}
	
	on_destroy = {
		if = {
			limit = {
				has_first_contact_dlc = yes
			}
			planet = {
				solar_system = {
					starbase = {
						fleet = {
							remove_modifier = covenant_detection_first_contact
						}
					}
				}
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @db3_minerals
			exotic_gases = @db3_rare_cost
		}
		upkeep = {
			energy = @db3_upkeep
			exotic_gases = @db3_rare_upkeep
		}
	}

	triggered_desc = {
		text = job_telepath_effect_desc
	}
}
