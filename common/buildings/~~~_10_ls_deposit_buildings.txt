## Alien Zoo (Tier 1)
# Merge between Sort Buildings, OSA
building_xeno_zoo = {
	base_buildtime = @db1_time
	position_priority = 3071000 # Sort Buildings
	base_cap_amount = 1
	category = amenity

    # Building Tooltip
    triggered_desc = {
        text = building_xeno_zoo_tt
    }

	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
		}
		planet = {
			OR = {
				has_deposit = d_alien_pets_deposit
				AND = {
					exists = owner
					owner = { has_policy_flag = pre_sapients_protect }
					any_owned_pop = {
						is_sapient = no
					}
				}
				any_owned_pop = {
					is_livestock = yes
				}
			}
		}
	}

	destroy_trigger = {
		OR = {
			AND = {
				exists = owner
				owner = {
					is_regular_empire = no
				}
			}
			planet = {
				NOR = {
					has_deposit = d_alien_pets_deposit
					AND = {
						exists = owner
						owner = { has_policy_flag = pre_sapients_protect }

							is_sapient = no
