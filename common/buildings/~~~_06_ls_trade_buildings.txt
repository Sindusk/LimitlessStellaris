## Commercial Zones (Tier 1)
# Merge between Sort Buildings, OSA, Bug Branch, Real Space: Planetary Stations, StarTech, Ariphaos
building_commercial_zone = {
	base_buildtime = @db1_time
	position_priority = 4061000 # Sort Buildings
	base_cap_amount = 5 # OSA caps to 5 maximum

	exempt_from_ai_planet_specialization = no
	category = trade

    # Building Tooltip
    triggered_desc = {
        text = building_commercial_zone_tt
    }

	potential = {
		exists = owner
		owner = { # StarTech AI
			is_gestalt = no
			OR = {
				is_ai = no
				years_passed < 3
				prev = { str_uses_comzones = yes }
			}
		}
		NOT = { has_modifier = slave_colony }
		NOT = { has_building = building_low_tech_capital }
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_gestalt = yes }
			has_modifier = slave_colony
			AND = { # StarTech AI
				owner = { is_ai = yes }
				years_passed > 5
				str_uses_comzones = no 
			}
		}
	}

	allow = {
		hidden_trigger = { # StarTech AI
			OR = {
				owner = { is_ai = no }
				AND = {
					str_uses_comzones = yes
					free_amenities < 2
					OR = {
						str_is_col_research = no
						str_is_col_bureau = no
						free_amenities < -5
					}
				}
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @db1_minerals
		}
		upkeep = { # OSA
			energy = 2.5
			minerals = 2.5
		}
		produces = { # OSA
			trigger = {
				owner = {
					has_active_tradition = tr_commerce_3
				}
			}
			unity = 5
		}
	}

	planet_modifier = {
		job_merchant_add = 1 # OSA
		job_clerk_add = 3
		job_bugged_trader_add = 1 # Bug Branch
		trade_value_mult = 0.05 # OSA
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_merchant_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		text = job_merchant_effect_desc
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_gestalt = no
	}

	upgrades = {
		"building_commercial_megaplex"
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}
	
	triggered_desc = { # Bug Branch
		text = job_bugged_trader_effect_desc
	}
	
	# StarTech AI Resource Production
	ai_resource_production = {
		energy = 2
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 0
		trigger = {
			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
					job = servant
					value >= 10
				}
			}
		}
	}
}

		"tech_interstellar_economics"
	}

		is_gestalt = no

		"building_commercial_trading_terminal" # OSA
	}

	ai_resource_production = {
		energy = 4

			always = yes
		}

			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
				job = servant
					value >= 10
				}
			}

## Planetary Stock Exchange (Tier 1) [Bug Branch]
# --- Here as a reference until further adjustments are made.
building_bugged_planetary_stock_exchange = {
	base_buildtime = 480
	base_cap_amount = 1

	category = trade

	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		has_major_upgraded_capital = yes
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
	}

	planet_modifier = {
		job_merchant_add = 1
		trade_value_mult = 0.05
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_civic = civic_merchant_guilds
					has_civic = civic_bugged_state_monopoly
				}
			}
		}
		modifier = {
			job_merchant_add = 1
			trade_value_mult = 0.05
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 600
			rare_crystals = 100
		}
		upkeep = {
			energy = 5
			rare_crystals = 2
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	upgrades = {
		"building_galactic_stock_exchange"

	}
	triggered_desc = {
		text = job_merchant_effect_desc
	}

	ai_resource_production = {
		energy = 3

			always = yes
		}
	}
	ai_resource_production = {
		energy = 5
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
