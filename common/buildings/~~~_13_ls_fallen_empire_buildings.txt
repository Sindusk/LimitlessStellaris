# Merge between Sort Buildings, Real Space: Planetary Stations, ACOT
building_affluence_center = {
	base_buildtime = 480
	can_build = yes
	can_demolish = yes
	position_priority = 42000 # Sort Buildings
	
    # Building Tooltip
    triggered_desc = {
        text = building_affluence_center_tt
    }

	potential = {
		custom_tooltip = {
			exists = owner
			OR = {
				owner = { is_fallen_empire = yes }
				owner = {
					has_country_flag = affluence_center_can_build
				}
				has_planet_flag = fallen_empire_world
			}
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = ancrel.10004.failtext
			owner = {
				check_variable = { which = affluence_center_build_count value >= 1 }
			}
		}
	}

	planet_modifier = {
		planet_amenities_add = 50
	}

	resources = {
		category = planet_buildings
		produces = {
			consumer_goods = 200
			unity = 100
			society_research = 100
		}
		upkeep = {
		}
	}
	on_queued = {
		owner = {
			subtract_variable = { which = affluence_center_build_count value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = affluence_center_build_count value = 1 }
		}
	}
	ai_weight = {
		weight = 1
			modifier = {
				factor = 50
				owner = {
				resource_income_compare = {
					resource = consumer_goods
					value < 10
					}
				}
			}
			modifier = {
				factor = 0.05
				owner = {
				resource_income_compare = {
					resource = consumer_goods
					value > 40
					}
				}
			}
			modifier = {
				factor = 0
				owner = {
					not = {
						has_country_flag = just_destroyed_awakened_consgoods
					}	
					resource_income_compare = {
						resource = consumer_goods
						value > 150
					}
					resource_stockpile_compare = {
						resource = consumer_goods
						value > 5000
					}
				}
			}
		}
		on_destroy = {
			owner = {
			set_timed_country_flag = {
				flag = just_destroyed_awakened_consgoods
				days = 40
			} }
		}		


	upgrades = {
		building_ae_affluence_center
	}
}

        text = building_fe_dome_tt
    }

	potential = {
		custom_tooltip = {
			exists = owner
			OR = {
				owner = { is_fallen_empire = yes }
				owner = {
					has_country_flag = skydome_can_build
				}
				has_planet_flag = fallen_empire_world
			}
		}
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
	}

	allow = {
		custom_tooltip = {
			fail_text = ancrel.10004.failtext
			owner = {
				check_variable = { which = skydome_build_count value >= 1 }
			}
		}
	}
	
	planet_modifier = {
		planet_max_buildings_add = 2
		planet_max_districts_add = 2
		planet_housing_add = 50
		planet_amenities_add = 50
	}
	
	resources = {
		category = planet_buildings
		upkeep = {
		}
	}
	on_queued = {
		owner = {
			subtract_variable = { which = skydome_build_count value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = skydome_build_count value = 1 }
		}
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0.05
			free_amenities > 20
		}
		modifier = {
			factor = 50
			free_amenities < -10
		}
		modifier = {
			factor = 0
			owner = {
				not = { has_country_flag = just_destroyed_awakened_amenities } 
			}
			free_amenities > 50
		}
	}

	on_destroy = {
		owner = { 
			set_timed_country_flag = {
				flag = just_destroyed_awakened_amenities
				days = 40
			} 
		}		
	}

	upgrades = {
		building_ae_sky_dome
	}
}
