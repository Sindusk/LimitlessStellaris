## Luxury Residences (Tier 1)
# Merge between Sort Buildings, EUTAB, Venture Politics, Planetary Diversity: Planetary Habitats, Real Space: Planetary Stations, StarTech
building_luxury_residence = {
	base_buildtime = @db1_time
	position_priority = 4071000 # Sort Buildings

	category = amenity

    # Building Tooltip
    triggered_desc = {
        text = building_luxury_residence_tt
    }

	potential = {
		OR = {
			NOR = {
				is_planet_class = pc_habitat
				is_pd_habitat = yes # Planetary Diversity: Planetary Habitats
			}
			has_planet_flag = advanced_habitat
		}
		NOT = { has_modifier = slave_colony }
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
		exists = owner
		owner = {
			is_gestalt = no
			NOR = {
				has_valid_civic = civic_shared_burden
				has_valid_civic = civic_employee_ownership # Venture Politics
			}
		}
		NOT = { has_building = building_low_tech_capital }
	}

	allow = {
		hidden_trigger = { # StarTech AI
			OR = {
				owner = { is_ai = no }
				AND = {
					NOR = {
						has_building = building_luxury_residence
						has_building = building_research_lab_1
					}
					is_capital = no
					num_pops >= 13
					free_jobs > 2
					OR = {
						AND = {
							free_amenities < -4
							free_amenities > -9
							OR = {
								free_district_slots < 3
								num_pops < 16
							}
						}
						AND = {
							free_building_slots > 1
							free_amenities < -2
							free_amenities > -6
							free_district_slots = 0
							has_building = building_holo_theatres
						}
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				free_building_slots = 0
				free_jobs < -2 # StarTech AI
				free_housing > 10
				free_district_slots = 0 # StarTech AI
				owner = {
					is_ai = yes
				}
			}
			AND = { # StarTech AI
				owner = { lategame = no }
				num_buildings = {
					type = building_research_lab_1
					value > 1	
				}
				free_jobs < 1
				free_amenities > 6
				free_building_slots = 0
				owner = {
					is_ai = yes
				}
			}
			owner = { is_gestalt = yes }
			owner = { 
				OR = {
					has_valid_civic = civic_shared_burden
					has_valid_civic = civic_employee_ownership # Venture Politics
				}
			}
			has_modifier = slave_colony
		}
	}

	convert_to = {
		building_drone_storage
		building_hive_warren
		building_communal_housing
	}

	planet_modifier = {
		planet_housing_add = 3
		planet_amenities_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_domination_imperious_architecture
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = { # EUTAB
		potential = {
			exists = owner
			owner = {
				has_technology = tech_eutab_habitation_studies
			}
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @db1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	upgrades = {
		"building_paradise_dome"
	}
}

## Hive Warren (Tier 1)
# Merge between Sort Buildings, Real Space: Planetary Stations, StarTech
building_hive_warren = {
	base_buildtime = @db1_time
	position_priority = 4071000 # Sort Buildings

	category = amenity

    # Building Tooltip
    triggered_desc = {
        text = building_hive_warren_tt
    }

	potential = {
		exists = owner
		owner = { is_hive_empire = yes }
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
	}

	allow = {
		buildings_simple_allow = yes # StarTech AI
		hidden_trigger = { # StarTech AI
			OR = {
				owner = { is_ai = no }
				AND = {
					free_jobs > 0
					free_amenities < -2
					free_amenities > -8
					OR = {
						#free_amenities > -10 PM: -11 is LESS than -8, so this will never be true due to earlier check for > -8
						AND = {
							num_pops > 13
							free_housing < 15
						}
						AND = {
							num_pops > 20
							free_housing < 25
						}
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = { # StarTech AI
			AND = {
				planet = {
					free_building_slots = 0
					free_jobs < -3
					free_housing > 6
					free_district_slots = 0
					owner = {
						is_ai = yes
					}
				}
			}
			NOR = {
				owner = { is_hive_empire = yes }
			}
		}
	}

	convert_to = {
		building_luxury_residence
		building_drone_storage
		building_communal_housing
	}

	planet_modifier = {
		planet_housing_add = 3
		planet_amenities_add = 5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @db1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	upgrades = {
		"building_expanded_warren"
	}
	
	# StarTech AI Resource Production
	ai_resource_production = {
		alloys = 10
		energy = 10
		trigger = {
			num_pops > 15
			free_amenities < -8
			free_housing < 20
			free_jobs > 1
			free_building_slots > 0
			or = {
				has_planet_flag = str_good_habitability
				has_planet_flag = str_acceptable_habitability
			}
			owner = {
				has_country_flag = str_has_room_to_grow
				resource_income_compare = {
					resource = alloys
					value > 20
				}
			}
		}
	}
}

## Holo-Theatres (Tier 1)
# Merge between Sort Buildings, Real Space: Planetary Stations, StarTech
building_holo_theatres = {
	base_buildtime = @db1_time
	position_priority = 4072000 # Sort Buildings

	category = amenity

    # Building Tooltip
    triggered_desc = {
        text = building_holo_theatres_tt
    }

	potential = {
		NOT = {	has_modifier = slave_colony	}
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
		owner = {
			is_regular_empire = yes
			NOT = { has_valid_civic = civic_dystopian_society }
		}
		OR = {
			owner = {
				NOT = { is_ai = yes }
			}
			AND = {
				NOT = { has_available_jobs = entertainer }
				NOT = { has_forbidden_jobs = entertainer }
			}
		}
		NOT = { has_building = building_low_tech_capital }
	}

	allow = {
		hidden_trigger = { # StarTech AI
			OR = {
				owner = { is_ai = no }
				AND = {
					free_amenities < -5
					num_pops > 11
				}
				AND = {
					free_amenities < -4
					OR = {
						has_building = building_luxury_residence
						has_building = building_communal_housing
					}
				}
				AND = {
					owner = { has_origin = origin_default }
					is_capital = yes
					NOT = { has_building = building_holo_theatres }
					free_amenities < 2
					OR = {
						years_passed > 15
						free_amenities < -2
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			owner = { has_valid_civic = civic_dystopian_society }
			has_modifier = slave_colony
			AND = { # StarTech AI
				owner = { is_ai = yes }
				free_amenities > 30
				years_passed > 25
			}
