## Noble Estates (Tier 1)
# Merge between Sort Buildings, OSA, Bug Branch, Real Space: Planetary Stations, StarTech
building_noble_estates = {
	base_buildtime = @db2_time
	icon = building_palace
	base_cap_amount = 1
	position_priority = 2021000 # Sort Buildings
	
    # Building Tooltip
    triggered_desc = {
        text = building_noble_estates_tt
    }

	category = government

	potential = {
		exists = owner
		planet = {
			or = {
				has_building = building_noble_estates
				owner = { is_ai = no }
			}
		}
		owner = { is_bugged_governmental_elitist = yes } # Bug Branch
		NOT = { has_modifier = slave_colony }
		NOT = { has_planet_flag = planetary_station } # Real Space: Planetary Stations
	}

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { # StarTech AI
			exists = owner
			OR = {
				owner = { is_ai = no }
				num_pops > 40
			}
		}
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_noble_estates
						is_starbase_building_building = ring_noble_estates
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		NOT = {
			owner = { is_bugged_governmental_elitist = yes } # Bug Branch
		}
	}

	planet_modifier = {
		job_noble_add = 2 # OSA 1 -> 2
		planet_housing_add = 5 # OSA 1 -> 5
	}

	triggered_planet_modifier = { # Bug Branch
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_bugged_peerage_assembly
			}
		}
		job_noble_add = 1
		planet_housing_add = 1
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @db1_minerals
		}
		upkeep = {
			energy = 2.5
			minerals = 2.5
		}
		produces = { # OSA
			trigger = {
				owner = {
					has_active_tradition = tr_order_4
				}
			}
			unity = 20
		}
	}

	triggered_desc = {
		text = job_noble_effect_desc
	}
	
	# StarTech AI Fake Job Manipulation
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_ai = yes }
			NOT = { has_building = building_noble_estates }
		}
		modifier = {
			job_starnet_spice_melange_farmer_add = 1
			job_starnet_spice_melange_antifarmer_add = -1
		}
	}
	ai_resource_production = {
		trigger = { num_pops > 40 }
		alloys = 15
		energy = 10
		minerals = 15
	}
}
