## Scrap Refinery (Tier 1)
# Add tooltip, Sort Buildings
building_low_tech_scrap_refinery = {
	can_build = no
	can_demolish = yes
	can_be_ruined = yes
	position_priority = 1041000 # Sort Buildings

	category = manufacturing

    # Building Tooltip
    triggered_desc = {
        text = building_low_tech_scrap_refinery_tt
    }

	empire_limit = {
		base = 1
	}

	base_buildtime = @b0_time

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b0_minerals
		}
		upkeep = {
			energy = @b0_upkeep
		}
	}

	potential = {
		exists = owner
		owner = {
			has_origin = origin_broken_shackles
		}
		has_deposit = d_crashed_slaver_ship
	}

	destroy_trigger = {
		exists = owner
		owner = {
			NOT = { has_origin = origin_broken_shackles }
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_lithoid_empire = yes
			}
		}
		job_low_tech_miner_add = 2
	}

	planet_modifier = {
		job_low_tech_laborer_add = 2
		job_low_tech_miner_add = 2
	}

	triggered_desc = {
		text = job_low_tech_miner_effect_desc
	}

	triggered_desc = {
		text = job_low_tech_laborer_effect_desc
	}
}

	can_build = no
	can_demolish = no
	can_be_ruined = no
	can_be_disabled = no
	position_priority = 0

	category = government

    # Building Tooltip
    triggered_desc = {
        text = building_low_tech_capital_tt
    }

	empire_limit = {
		base = 1
	}

	potential = {
		exists = owner
		owner = {
			has_origin = origin_broken_shackles
		}
		OR = {
			has_deposit = d_crashed_slaver_ship
			has_deposit = d_crashed_slaver_ship_memorial
		}
	}

	convert_to = {
		building_machine_capital
		building_hive_capital
	}

	planet_modifier = {
		job_colonist_add = 2
		job_low_tech_warrior_add = 1
		planet_housing_add = 4
		planet_amenities_add = 7
		planet_max_buildings_add = 1
	}

	upgrades = {
		building_capital
	}

	triggered_desc = {
		text = job_colonist_effect_desc
	}

	triggered_desc = {
		text = job_low_tech_warrior_effect_desc
	}
}
