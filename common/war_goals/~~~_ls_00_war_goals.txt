## Independence [Vanilla]
# Merge between VER, Organic Zealots, MECR, StarTech
wg_independence = {
	casus_belli = cb_subject

	hide_if_no_cb = yes
	set_defender_wargoal = wg_assert_overlordship

	potential = {
		always = yes
	}
	possible = {
		hidden_trigger = { exists = from } # Organic Zealots
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
		custom_tooltip = { # MECR
			fail_text = peace_windows_project_active
			NAND = {
				is_active_resolution = resolution_custodian_peace_windows_project
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	on_status_quo = {
		custom_tooltip = wg_independence_status_quo
		hidden_effect = {
			fromfrom = {
				every_war_participant = {
					limit = {
						fromfrom = {
							is_war_participant = {
								who = prev
								side = Root
							}
						}
						has_overlord = Root.from
					}
					set_subject_of = {
						who = none
					}
					fire_on_action = { # Organic Zealots
						on_action = on_becoming_independent
						scopes = {
							from = root.from
						}
					}
				}
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_independence_accept
		hidden_effect = {
			fromfrom = {
				every_war_participant = {
					limit = {
						fromfrom = {
							is_war_participant = {
								who = prev
								side = Root
							}
						}
						has_overlord = Root.from
					}
					set_subject_of = {
						who = none
					}
					fire_on_action = { # Organic Zealots
						on_action = on_becoming_independent
						scopes = {
							from = root.from
						}
					}
					every_system_within_border = {
						limit = {
							Root.from = { has_claim = prev }
						}
						remove_claims = { who = Root.from }
					}
				}
			}
		}
	}

	on_wargoal_set = {
		if = { # VER completely rewrites this section
			limit = {
				is_tributary = yes
				NOT = { has_country_flag = new_tributary }
			}
			random_war = {
				limit = { using_war_goal = { type = wg_independence owner = prev } }
				every_war_participant = { prev = { remove_war_participant = prev } }
			}
			country_event = { id = veru_l.3010 }
		}
		else = {
			every_system_within_border = {
				limit = { NOT = { from = { has_claim = prev } } }
				add_claims = { who = from }
			}
		}
	}

	ai_weight = {
		weight = 100
	}

	should_ignore_federations = yes
}

					has_country_flag = released_empire
				}
			}

			last_created_country = {

					set_merchant_government_effect = yes
				}
			}
		}

				}
			}

			last_created_country = {
				set_origin_effect = yes
				join_alliance = { who = root override_requirements = yes }
			}

			if = {
				limit = {
					root = { is_megacorp = yes }
					last_created_country = { is_megacorp = yes }
				}
				last_created_country = {
					set_merchant_government_effect = yes
				}
			}
		}
	}

		weight = 10
	}
