## Standard Pop Modifiers
# Merge between DFEM, DTM
triggered_planet_modifier = { # DTM
	potential = {
		valuable_secretions_check = yes
		can_generate_trade_value = yes
	}
	trade_value_add = 0.4
}

	potential = {
		has_trait = trait_robot_surveilance_protocol
		is_being_purged = no
	}
	planet_crime_add = -0.5
}

	potential = {
		has_trait = trait_robot_poor_security

	}

	potential = {
		has_trait = trait_robot_poor_security_machine

	}

}

triggered_planet_modifier = { # DTM
	potential = {
		has_trait = trait_baby_crazy
		NOT = {
			has_migration_control = { type = migration_control_yes }
		}
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
	}
	planet_emigration_push_add = 2
	mult = value:pop_growth_suppression
}

}
