### Triggered Planet Modifiers
## Expanded Megastructures and Technology - Engineered Evolution
# Additional organic pop assembly for Engineered Evolution ascension perk.
triggered_planet_modifier = { # Expanded Megastructures and Technology
	potential = {
		exists = owner
		owner = {
			has_ascension_perk = ap_engineered_evolution
		}
	}
	modifier = {
		planet_pop_assembly_organic_add = 1 # LS 2 -> 1 [Balance]
	}
}
