# Merge between Planetary Diversity: Unique Worlds

## Triggered Planet Modifiers
# Vanilla - Non-catalytic hive jobs. Added PD conditions
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_hive_empire = yes
			is_catalytic_empire = no
		}
		NOT = { # PD: Unique Worlds
			OR = {
				is_planet_class = pc_crucible 
				is_planet_class = pc_shroud_crucible 
				is_planet_class = pc_clean_crucible
			}
		}
	}
	modifier = {
		job_alloy_drone_add = $basic$
	}
}
