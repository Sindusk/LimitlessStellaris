### Triggered Planet Modifier
# Miner Jobs
triggered_planet_modifier = {
	potential = {
		NOT = { is_planet_class = pc_shattered_ring_habitable }
		exists = owner
		owner = {
			is_regular_empire = yes
			is_fallen_empire_spiritualist = no
		}
	}
	modifier = {
		job_miner_add = $basic$
	}
}
