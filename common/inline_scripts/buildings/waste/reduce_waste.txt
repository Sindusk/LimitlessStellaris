## Ls - Reduction to non-gestalt waste.
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = { is_gestalt = no }
	}
	modifier = {
		pop_category_rulers_waste_produces_add = $per_pop$
		pop_category_specialists_waste_produces_add = $per_pop$
		pop_category_workers_waste_produces_add = $per_pop$
	}
}
