## Orbital Ring - Habitation
# Expanded Starbases version
orbital_ring_habitation = {
	icon = "GFX_orbitalring_module_habitation"
	section = "HABITAT_ORBITAL_RING_SECTION"
	construction_days = 360

	potential = {
		exists = owner
		owner = { has_technology = tech_housing_2 }
		is_orbital_ring = yes
	}

	show_in_tech = "tech_housing_2"

	resources = {
		category = starbase_modules
		cost = {
			alloys = 200
			influence = 25
		}

		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_max_districts_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet
			planet = {
				exists = owner.overlord
				has_holding = {
					holding = holding_orbital_assembly_complex
					owner = owner.overlord
				}
			}
		}
		planet_max_buildings_add = 0.5
	}

	replaceable = {
		custom_tooltip = {
			fail_text = orbital_ring_habitation_districts_maxed_out
			OR = { # to keep tooltip from showing up when no module is built yet
				NOT = { has_starbase_module = orbital_ring_habitation }
				AND = {
					has_starbase_module = orbital_ring_habitation
					planet = {
						num_free_districts = { type = any value >= 1 }
					}
				}
			}
		}
	}

	ai_build_at_chokepoint = no
	ai_build_outside_chokepoint = yes
	ai_weight = {
		weight = 100
		modifier = {
			factor = 3
			exists = planet
			planet = {
				num_free_districts = { type = any value < 1 }
			}
		}
		modifier = {
			factor = 2
			exists = planet
			planet = {
				OR = {
					has_designation = col_mining
					has_designation = col_farming
					has_designation = col_generator
					has_designation = col_industrial
					has_designation = col_foundry
					has_designation = col_factory
					has_designation = col_rural
					has_designation = col_rural_gestalt
				}
			}
		}
	}
}

## Orbital Ring - Shipyard
# Expanded Starbases version
orbital_ring_shipyard = {
	icon = "GFX_starbase_shipyard"
	section = "SHIPYARD_ORBITAL_RING_SECTION"
	construction_days = 180
	initial = yes

	potential = {
		is_orbital_ring = yes
	}

	resources = {
		category = starbase_modules
		cost = {
			alloys = 50
		}

		upkeep = {
			energy = 1
		}
	}

	station_modifier = {
		starbase_shipyard_capacity_add = 1
	}

	triggered_station_modifier = {
		potential = {
			exists = planet
			planet = {
				exists = owner.overlord
				has_holding = {
					holding = holding_orbital_assembly_complex
					owner = owner.overlord
				}
			}
		}
		starbase_shipyard_capacity_add = 1
	}


	ai_build_at_chokepoint = no
	ai_build_outside_chokepoint = yes
	ai_weight = {
		weight = 100
		modifier = {
			factor = 100
			owner = {
				OR = {
					count_owned_starbase = {
						count = 0
						limit = { has_starbase_module = orbital_ring_shipyard }
					}
					AND = {
						num_starbases >= 4
						count_owned_starbase = {
							count < 2
							limit = { has_starbase_module = orbital_ring_shipyard }
						}
					}
					AND = {
						num_starbases >= 9
						count_owned_starbase = {
							count < 3
							limit = { has_starbase_module = orbital_ring_shipyard }
						}
					}
					AND = {
						num_starbases >= 15
						count_owned_starbase = {
							count < 4
							limit = { has_starbase_module = orbital_ring_shipyard }
						}
					}
				}
			}
		}
		modifier = {
			factor = 3
			has_starbase_module = orbital_ring_shipyard
			count_starbase_modules = {
				count < 2
				type = orbital_ring_shipyard
			}
		}
		modifier = {
			factor = 3
			has_starbase_module = orbital_ring_shipyard
			count_starbase_modules = {
				count < 4
				type = orbital_ring_shipyard
			}
		}
	}
}
