	icon = "GFX_starbase_shipyard"
	section = "ASSEMBLYYARD_STARBASE_SECTION"
	construction_days = 180
	initial = yes

	potential = {
		is_orbital_ring = no
	}

	resources = {
		category = starbase_modules
		cost = {
			alloys = 150 # OSA 50 -> 150
		}
		produces = { # OSA
			trigger = {
				exists = owner
				owner = {
					has_active_tradition = tr_ingenuity_5
				}
			}
			engineering_research = 2
		}

		upkeep = {
			energy = 1
			minerals = 1 # OSA
		}
	}

	station_modifier = {
		ship_hull_mult = 0.25 # OSA
		ship_armor_mult = 0.25 # OSA
		ship_shield_mult = 0.25 # OSA
		starbase_shipyard_capacity_add = 1
	}
	triggered_station_modifier = {
		potential = { has_starbase_building = irassian_naval_yards }
		starbase_shipyard_capacity_add = 1
	}

	ai_build_at_chokepoint = yes # StarTech
	ai_build_outside_chokepoint = yes
	ai_weight = {
		weight = 300 # StarTech 100 -> 300
		
		modifier = { # StarTech AI
			factor = 0
			not = { has_starbase_module = shipyard }
			owner = { needs_shipyard = no }
		}
		
		modifier = { # StarTech AI
			factor = 0
			has_starbase_module = anchorage
			not = { has_starbase_module = shipyard }
		}
	}
}

trading_hub = {
	icon = "GFX_starbase_trading_hub"
	section = "REFINERY_STARBASE_SECTION"
	construction_days = 180
	initial = yes

	potential = {
		exists = owner
		owner = {
			is_gestalt = no
		}
		is_orbital_ring = no
	}

	resources = {
		category = starbase_modules
		cost = {
			alloys = 150 # OSA 50 -> 150
		}
		upkeep = {
			energy = 1
		}
	}

	station_modifier = {
		starbase_trade_collection_range_add = 1
		trade_value_add = 2 # Expanded Starbases
	}

	triggered_station_modifier = { # Expanded Starbases
		potential = {
			solar_system = {
				has_star_flag = enclave
				any_ship_in_system = {
					exists = owner
					owner = {
						has_country_flag = trader_enclave_country
					}
				}
			}		
		}

		trade_value_add = 2
	}

	triggered_station_modifier = {
		potential = {
			hidden_trigger = { exists = this }
			owner = {
				has_councilor = { COUNCILOR = "councilor_trading_posts" }
			}
		}
		trade_value_add = 0.5
		mult = value:councilor_level|TYPE|councilor_trading_posts|
	}

	triggered_station_modifier = { # Bug Branch
		potential = {
			exists = owner
			owner = {
				has_policy_flag = bugged_civilian_telecom
			}
		}
		trade_value_add = 2
	}

	ai_build_at_chokepoint = no # StarTech AI
	ai_weight = {
		weight = 0 # StarTech disables construction
	}
}
