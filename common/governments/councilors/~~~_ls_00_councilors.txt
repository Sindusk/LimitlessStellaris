councilor_research = {
	leader_class = { scientist governor }

	possible = {
		is_gestalt = no
	}

	is_leader_possible = {
		OR = {
			leader_class = scientist
			leader_class = governor
		}
	}
	modifier = {
		all_technology_research_speed = 0.02
		assist_research_mult = 0.02
	}
	icon = "GFX_icon_councilor_research"
	required = yes
}

councilor_agrarian_idyll = {
	possible = { # Subtle Polish
		OR = {
			AND = {
				is_scope_valid = yes
				OR = {
					NOT = { exists = owner_species }
					is_lithoid_empire = no
				}
			}
			AND = {
				is_scope_valid = no
				NOT = { local_human_species_class = LITHOID }
			}
		}
	}
	leader_class = { general governor }
	is_leader_possible = {
		OR = {
			leader_class = general
			leader_class = governor
		}
	}
	civic = civic_agrarian_idyll
	modifier = {
		planet_farmers_produces_mult = 0.05
		planet_farmers_society_research_produces_add = 0.2 # OSA
	}
}
