# Peacekeepers 
civic_pk = {
	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } } # Must be non-gestalt to have pacifist or fanatic pacifist ethics
	}
	ai_playable = { always = no }
	possible = {
		civics = {
			NOR = {
				value = civic_warlord
				value = civic_hoi
				value = civic_regime
				value = civic_hive_devouring_swarm
				value = civic_fanatic_purifiers
				value = civic_machine_terminator
			}
		}
		ethics = {
			OR = {
				value = ethic_pacifist
				value = ethic_fanatic_pacifist
			}
		}
	}		
	modification = no
	random_weight = { base = 1 }
	modifier = {
		starbase_modules_cost_mult = -0.1
		shipclass_starbase_damage_mult = 0.1
		shipclass_starbase_hull_mult = 0.1
		country_starbase_capacity_add = 10
		ships_upkeep_mult = 0.05
	}
}
