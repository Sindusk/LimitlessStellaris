# Merge between OSA, Bug Branch, EMEX
civic_franchising = {
	potential = {
        OR = {
			authority = { # Bug Branch
				OR = {
					value = auth_bugged_corporate_democratic
					value = auth_corporate
					value = auth_bugged_corporate_imperial
				}
			}
            civics = { value = civic_galactic_sovereign_megacorp }
        }
	}
	random_weight = { base = @civic_default_random_weight }
	ai_weight = { 
		base = @ai_civic_default_base_weight 
		modifier = {
			factor = @ai_civic_personality_match_factor
			OR = {
				has_ai_personality = ruthless_capitalists
				has_ai_personality = hegemonic_imperialists
				has_ai_personality = slaving_despots
				has_ai_personality = federation_builders
				has_ai_personality = fanatic_befrienders
			}
		}
		modifier = {
			factor = @ai_civic_personality_mismatch_factor
			OR = {
				has_ai_personality = xenophobic_isolationists
			}
		}
	}
	description = "civic_tooltip_franchising_effects"
	modifier = {
		country_civics_influence_produces_add = 1 # OSA
		country_subject_power_penalty_mult = -0.33 # Bug Branch
		empire_size_branch_office_mult = -0.25
		planet_branch_offices_cost_mult = -0.15 # EMEX
	}
}
