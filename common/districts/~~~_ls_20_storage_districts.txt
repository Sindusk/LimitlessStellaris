district_planet_storage = {
	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
			#-----PD COMP-----#
			uses_district_set = pdflorafort
			uses_district_set = pdplanethab
			uses_district_set = pdcrucible
			uses_district_set = pdarchive
		}
	}

	potential = {
		exists = owner
		owner = { is_ai = no }
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
			#-----PD COMP-----#
			uses_district_set = pdflorafort
			uses_district_set = pdplanethab
			uses_district_set = pdcrucible
			uses_district_set = pdarchive
		}
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 10
			alloys = 1
		}
	}

	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 1
	convert_to = {
		district_ecum_storage
	}

	planet_modifier = {
		planet_housing_add = 1
	}
	
	#####-----NORMAL-----#####
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			job_yanz_storage_specialist_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
					has_active_building = building_storage_bureau
		}
		modifier = {
			job_yanz_storage_specialist_add = 1
		}
	}
	#####-----HIVE-----#####	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_authority = "auth_hive_mind"
			}
		}
		modifier = {
			job_yanz_gestalt_storage_drone_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_authority = "auth_hive_mind"
			}
					has_active_building = building_storage_bureau
		}
		modifier = {
			job_yanz_gestalt_storage_drone_add = 1
		}
	}
	#####-----MACHINE-----#####
