# Merge between OSA, Bug Branch, Planetary Diversity, Lithoid Supremacy, StarTech, Subtle Polish, Ariphaos
district_city = {
	base_buildtime = 480
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	show_on_uncolonized = {
		exists = from
		from = { is_gestalt = no } # Merged
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
			uses_district_set = pdnecroplanet # Planetary Diversity
			uses_district_set = pdscrapyard # Planetary Diversity
		}
		NOT = { has_deposit = d_crashed_slaver_ship } # Subtle Polish
		exists = from # Subtle Polish
		from = { # Subtle Polish
			OR = {
				is_regular_empire = yes
				AND = {
					is_country_type = primitive
					is_hive_empire = no
				}
			}

	}

	potential = {
		OR = {
			NOT = { exists = owner } #this condition is needed for galaxy generation
			AND = {
				exists = owner

					OR = {
						is_regular_empire = yes
						AND = {
							is_country_type = primitive
							is_hive_empire = no
						}
					}
				}
			}
		}
		NOT = { has_deposit = d_crashed_slaver_ship }
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
			uses_district_set = shattered_ring_world
			uses_district_set = pdnecroplanet # Planetary Diversity
			uses_district_set = pdscrapyard # Planetary Diversity
		}
		NOT = { # OSA
			has_planet_flag = osa_military
		}
	}

	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		str_city_district_allow = yes # StarTech AI
		OR = { # Ariphaos 
			NOT = { exists = owner }
			AND = {
				exists = owner
				OR = {
					owner = { is_ai = no }
					AND = {
						num_districts = { type = district_city value < 3 }
					}

			energy = 2
		}
