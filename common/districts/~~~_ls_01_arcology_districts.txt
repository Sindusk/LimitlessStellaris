## Leisure
# Merge between OSA, Venture Politics, Planetary Diversity, StarTech, Subtle Polish
district_arcology_leisure = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		OR = { # Planetary Diversity
			uses_district_set = city_world
			uses_district_set = pdecocity
			uses_district_set = pdmilarc
			uses_district_set = pdcorpoarc
			uses_district_set = pdtechnoarch
		}
		exists = from
		from = {
			is_regular_empire = yes
			NOT = { has_civic = civic_dystopian_society }
		}
	}

	potential = {
		OR = { # Planetary Diversity
			uses_district_set = city_world
			uses_district_set = pdecocity
			uses_district_set = pdmilarc
			uses_district_set = pdcorpoarc
			uses_district_set = pdtechnoarch
		}
		exists = owner
		owner = {
			is_gestalt = no # Merged
			NOT = { has_civic = civic_dystopian_society }
			NOT = { has_valid_civic = civic_urban_idyll } # Venture Politics
		}
	}

	allow = {
		str_habitat_culture_allow = yes # StarTech AI
	}

	conversion_ratio = 1
	convert_to = {
		district_arcology_organic_housing
		district_city
		district_nexus
		district_hive
		district_arcology_housing
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
			exotic_gases = 50
		}
		upkeep = {
			energy = 5
			exotic_gases = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 10
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no # Merged
				NOR = {
					has_valid_civic = civic_warrior_culture
					has_valid_civic = civic_corporate_mercenary_contractors # OSA
				}
			}
		}
		modifier = {
			job_entertainer_add = 3 # Subtle Polish 6 -> 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no # Merged
				OR = {
					has_valid_civic = civic_warrior_culture
					has_valid_civic = civic_corporate_mercenary_contractors # OSA
				}
			}
		}
		modifier = {
			job_duelist_add = 3 # Subtle Polish 6 -> 3
		}
	}
	
	triggered_planet_modifier = { # Subtle Polish
		potential = { always = yes }
		# uses triggered_planet_modifier to ensue it appears under the entertainer/duelist jobs
		job_culture_worker_add = 3
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_warrior_culture
					has_valid_civic = civic_corporate_mercenary_contractors # OSA
				}
			}
		}
		text = job_duelist_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				NOR = {
					has_valid_civic = civic_warrior_culture
					has_valid_civic = civic_corporate_mercenary_contractors # OSA
				}
			}
		}
		text = job_entertainer_effect_desc
	}
	
	triggered_desc = { # Subtle Polish
		text = job_culture_worker_effect_desc
	}
